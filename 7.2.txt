#include <stdio.h>

void matrix_multiplication(int a[3][3], int b[3][3], int result[3][3]) {
   int i, j, k;

   for (i = 0; i < 3; i++) {
      for (j = 0; j < 3; j++) {
         result[i][j] = 0;

         for (k = 0; k < 3; k++) {
            result[i][j] += a[i][k] * b[k][j];
         }
      }
   }
}

void print_matrix(int matrix[3][3]) {
   int i, j;

   for (i = 0; i < 3; i++) {
      for (j = 0; j < 3; j++) {
         printf("%d ", matrix[i][j]);
      }
      printf("\n");
   }
}

int main() {
   int a[3][3] = {{1, 2, 3}, {4, 5, 6}, {7, 8, 9}};
   int b[3][3] = {{10, 11, 12}, {13, 14, 15}, {16, 17, 18}};
   int result[3][3];

   matrix_multiplication(a, b, result);

   printf("Resultant Matrix:\n");
   print_matrix(result);

   return 0;
}
